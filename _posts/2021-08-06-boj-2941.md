#include <iostream>
#include <vector>

using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL); cout.tie(NULL);

    vector<string> vec = {"c=", "c-", "dz=", "d-", "lj", "nj", "s=", "z="};
    int index = 0;
    string S;
    cin >> S;
    for(int i=0; i<vec.size(); i++){
        index = S.find(vec[i]);
        for(auto i in S)
            if (S.find(vec[i])) S.replace(index, vec[i].length(), "#");
    }
    cout << S.length();
}
