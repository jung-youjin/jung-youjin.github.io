#include <iostream>

using namespace std;

int A[11][100001];
int temp[100001];
int ret[2]={0};

int* getMax(int a, int b, int* temp) { 
    if (temp[a]>=temp[b]) {
        ret[0] = a;
        ret[1] = temp[a];
    }
    return ret;
}

int main() {
    ios::sync_with_stdio(false);
    cout.tie(NULL); cin.tie(NULL);

    int T, Ai, N;
    cin >> T;
    if (T>10 || T<1) return 0;
    for (int i=0; i<T; i++) {
        int num;
        cin >> N;
        if (N<1 || N>100000) return 0;
        for (int j=0; j<N; j++) {
            cin >> Ai;
            if (Ai<1 || Ai>1000000000) return 0;
            A[i][j] = Ai;
        }
    }

    int comp = 0;
    
    for (int i=0; i<T; i++) {
        for (int k=0; k<=N-3; k++) {
            temp[k] = A[i][k] + A[i][k+1] + A[i][k+2];
            int* result = getMax(comp, k, temp);
            comp = result[0];
        }
        cout << A[i][comp] << " " << A[i][comp+1] << " " << A[i][comp+2] << " " << endl;
    }
    return 0;
}
